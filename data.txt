create table Users (
	id serial,
	name varchar(50) not null,
	last_name varchar(50) not null, 
	email varchar(50),
	password int not null
)

alter table Users
alter column password type varchar(256)

create table restaurants (
	id serial,
	name varchar(50) not null,
	address varchar(50) not null, 
	latitude bigint,
	longitude bigint,
	phone int,
	working_hours int,
	availability boolean,
	rate int,
	city varchar(50),
)
